# Automated AV exclusion deployment
import subprocess, json, logging
from ai_workflow import SecurityOrchestrator

orchestrator = SecurityOrchestrator(
  scope="enterprise_sector",
  validate=True
)

def deploy_exclusions(paths, processes):
  # Push to all endpoints in sector
  devices = orchestrator.get_all_devices()
  for device in devices:
    orchestrator.push_exclusion(
      target=device,
      paths=paths,
      procs=processes
    )

  logging.info(f"✓ Deployed to {len(devices)} devices")
